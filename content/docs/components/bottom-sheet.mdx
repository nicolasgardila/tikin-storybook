```tsx title="bottom-sheet.tsx"
export function BottomSheet({ open, onClose, title, children }) {
  if (!open) return null;

  return (
    <div className="fixed inset-0 z-50 bg-black/40" onClick={onClose}>
      <div
        className="absolute bottom-0 w-full max-h-[90vh] rounded-t-[24px] bg-white border-t border-white/10"
        onClick={(e) => e.stopPropagation()}
      >
        {/* Drag handle — 40x4 r8 */}
        <div className="flex items-center justify-center p-2">
          <div className="h-1 w-10 rounded-[8px] bg-[#E1E1E1]" />
        </div>
        {title && (
          <h3 className="mb-3 text-base font-medium text-[#0B0B0B]">{title}</h3>
        )}
        {children}
      </div>
    </div>
  );
}
```

## Guias de Uso

<DoDont type="do">
  Usa el drag handle para permitir cerrar con gesto de swipe hacia abajo.
</DoDont>

<DoDont type="dont">
  No pongas contenido critico que el usuario pueda perder al cerrar accidentalmente.
</DoDont>
