```tsx title="breadcrumbs-default.tsx"
export function Breadcrumbs({ items }) {
  return (
    <nav className="flex items-center font-['Ubuntu_Sans'] text-sm">
      {items.map((item, i) => (
        <Fragment key={i}>
          {i > 0 && (
            <span className="text-[#999999] px-2">
              <ChevronRight size={12} />
            </span>
          )}
          {i === items.length - 1 ? (
            <span className="font-semibold text-[#0B0B0B]">{item.label}</span>
          ) : (
            <a href={item.href} className="text-[#0B0B0B] hover:underline">
              {item.label}
            </a>
          )}
        </Fragment>
      ))}
    </nav>
  );
}
```

```tsx title="breadcrumbs-truncated.tsx"
export function TruncatedBreadcrumbs({ items, maxItems = 3 }) {
  const truncated = items.length > maxItems;
  const visible = truncated
    ? [items[0], '...', ...items.slice(-2)]
    : items;

  return (
    <nav className="flex items-center font-['Ubuntu_Sans'] text-sm">
      {visible.map((item, i) => (
        <Fragment key={i}>
          {i > 0 && (
            <span className="text-[#999999] px-2">
              <ChevronRight size={12} />
            </span>
          )}
          {item === '...' ? (
            <span className="text-[#999999]">...</span>
          ) : i === visible.length - 1 ? (
            <span className="font-semibold text-[#0B0B0B]">{item.label}</span>
          ) : (
            <a href={item.href} className="text-[#0B0B0B] hover:underline">
              {item.label}
            </a>
          )}
        </Fragment>
      ))}
    </nav>
  );
}
```

## Guias de Uso

<DoDont type="do">
  Siempre incluye la pagina actual como ultimo item (no clickeable, con font-weight 600).
</DoDont>

<DoDont type="dont">
  No uses breadcrumbs con menos de 2 niveles de profundidad. No aporta valor si solo hay un nivel.
