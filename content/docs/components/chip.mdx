        <XIcon className="w-2.5 h-2.5" />
      </button>
    </span>
  );
}

```tsx title="chip-variants.tsx"
const statusColors = {
  nuevo:     { bg: '#C1F5C1', color: 'var(--preview-brand, #0B0B0B)' },
  exitosa:   { bg: '#CFF7D3', color: '#02542D' },
  pendiente: { bg: '#FDDC72', color: 'var(--preview-brand, #0B0B0B)' },
  fallido:   { bg: '#FDD3D0', color: '#900B09' },
  expirado:  { bg: '#E3E3E3', color: '#434343' },
};

export function StatusTag({ label, variant }: { label: string; variant: keyof typeof statusColors }) {
  const { bg, color } = statusColors[variant];
  return (
    <span style={{ backgroundColor: bg, color, borderRadius: '44px', padding: '4px 8px', fontSize: '12px', fontFamily: 'var(--font-ubuntu-sans-mono), monospace', letterSpacing: '0.6px', textTransform: 'uppercase', border: '0.5px solid rgba(0,0,0,0.08)' }}>
      {label}
    </span>
  );
}
```

### Interacciones

El componente incluye feedback tactil al presionar: escala sutil (`scale 0.97`) con reduccion de opacidad, y un release con transicion spring natural. Este efecto se desactiva automaticamente en estados `disabled` y `loading`.

## Guias de Uso

<DoDont type="do">
  Usar chips para filtros activos, tags, o selecciones multiples compactas. Son ideales para representar metadata o categorias.
</DoDont>

<DoDont type="dont">
  No usar chips como botones de accion principal. Son para metadata y categorias, no para acciones del flujo principal.
</DoDont>
