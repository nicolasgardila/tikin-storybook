```tsx title="popover.tsx"
export function Popover({ open, onClose, trigger, position = 'bottom', children }) {
  return (
    <div className="relative inline-block">
      {trigger}
      {open && (
        <>
          <div className="fixed inset-0" onClick={onClose} />
          <div
            className={`absolute z-50 min-w-[160px] max-w-[280px] rounded-xl border border-[#DEDEDE] bg-white shadow-[0_4px_12px_rgba(0,0,0,0.1)]`}
            style={{ top: 'calc(100% + 4px)', left: 0 }}
          >
            {children}
          </div>
        </>
      )}
    </div>
  );
}
```

## Guias de Uso

<DoDont type="do">
  Usa para menus contextuales con 3-8 opciones relacionadas.
</DoDont>

<DoDont type="dont">
  No pongas formularios o contenido complejo dentro de un popover.
</DoDont>
