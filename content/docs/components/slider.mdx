```tsx title="slider-default.tsx"
export function Slider({ value, min = 0, max = 100 }: SliderProps) {
  const percent = ((value - min) / (max - min)) * 100;
  return (
    <div style={{ position: 'relative', height: 20, display: 'flex', alignItems: 'center', width: '100%' }}>
      <div style={{ position: 'absolute', width: '100%', height: 4, backgroundColor: '#EEEBE6', borderRadius: 100 }} />
      <div style={{ position: 'absolute', height: 4, backgroundColor: 'var(--preview-brand, #0B0B0B)', borderRadius: 100, width: `${percent}%` }} />
      <div style={{ position: 'absolute', left: `${percent}%`, transform: 'translateX(-50%)', width: 20, height: 20, borderRadius: '50%', backgroundColor: 'var(--preview-surface, #FFFFFF)', border: '2px solid var(--preview-brand, #0B0B0B)', cursor: 'pointer' }} />
    </div>
  );
}
```

```tsx title="slider-range.tsx"
export function RangeSlider({ low, high, min, max }: RangeSliderProps) {
  const lowPercent = ((low - min) / (max - min)) * 100;
  const highPercent = ((high - min) / (max - min)) * 100;
  return (
    <div className="relative h-5 flex items-center w-full">
      <div className="absolute w-full h-2 bg-[#F6F6F6] rounded-full" />
      <div
        className="absolute h-2 bg-[#0B0B0B] rounded-full"
        style={{ left: `${lowPercent}%`, width: `${highPercent - lowPercent}%` }}
      />
      <div className="absolute w-5 h-5 rounded-full bg-white border-2 border-[#0B0B0B] -translate-x-1/2" style={{ left: `${lowPercent}%` }} />
      <div className="absolute w-5 h-5 rounded-full bg-white border-2 border-[#0B0B0B] -translate-x-1/2" style={{ left: `${highPercent}%` }} />
    </div>
  );
}
```

## Guias de Uso

<DoDont type="do">
  Mostrar el valor actual y los limites del rango para que el usuario sepa el contexto. Usar labels cuando la precision es importante.
</DoDont>

<DoDont type="dont">
  No usar slider para valores exactos donde la precision es critica. Usar input numerico en esos casos.
</DoDont>
