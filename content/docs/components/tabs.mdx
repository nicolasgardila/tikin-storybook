## Instalacion

```bash title="Terminal"
npm install @dstikin/ds
```

```tsx
import { Tabs } from '@dstikin/ds'
```

## Uso

```tsx title="tabs.tsx"
import { Tabs } from '@dstikin/ds'

export default function Example() {
  return (
    <Tabs
      variant="contained"
      tabs={['General', 'Seguridad', 'Notificaciones']}
      activeTab={0}
      onChange={(index) => console.log(index)}
    />
  )
}
```

```tsx title="tabs-contained.tsx"
export function TabsContained() {
  const [activeTab, setActiveTab] = useState(0);
  const tabs = ['General', 'Seguridad', 'Notificaciones'];

  return (
    <div className="inline-flex items-center bg-[#F6F6F6] rounded-xl p-1 gap-0.5">
      {tabs.map((tab, i) => (
        <button
          key={tab}
          onClick={() => setActiveTab(i)}
          className={`rounded-xl px-4 py-2 text-sm font-mono font-medium uppercase tracking-wider border-none cursor-pointer transition-colors ${
            activeTab === i
              ? 'bg-white text-[#0B0B0B] shadow-sm'
              : 'bg-transparent text-[#999999]'
          }`}
        >
          {tab}
        </button>
      ))}
    </div>
  );
}
```

```tsx title="tabs-underline.tsx"
export function TabsUnderline() {
  const [activeTab, setActiveTab] = useState(0);
  const tabs = ['Movimientos', 'Favoritos', 'Programados'];

  return (
    <div className="inline-flex items-center gap-6">
      {tabs.map((tab, i) => (
        <button
          key={tab}
          onClick={() => setActiveTab(i)}
          className={`py-2 text-sm font-mono font-medium uppercase tracking-wider bg-transparent border-none cursor-pointer transition-colors ${
            activeTab === i
              ? 'text-[#0B0B0B] border-b-2 border-[#0B0B0B]'
              : 'text-[#999999] border-b-2 border-transparent'
          }`}
        >
          {tab}
        </button>
      ))}
    </div>
  );
}
```

### Interacciones

El componente incluye feedback tactil al presionar: escala sutil (`scale 0.97`) con reduccion de opacidad, y un release con transicion spring natural. Este efecto se desactiva automaticamente en estados `disabled` y `loading`.

## Guias de Uso

<DoDont type="do">
  Usar Contained para secciones dentro de un card. Usar Underline para navegacion de pagina.
</DoDont>

<DoDont type="dont">
  No mezclar variantes Contained y Underline en el mismo contexto.
</DoDont>
