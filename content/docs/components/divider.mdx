## Instalacion

```bash title="Terminal"
npm install @dstikin/ds
```

```tsx
import { Divider } from '@dstikin/ds'
```

## Uso

```tsx title="divider.tsx"
import { Divider } from '@dstikin/ds'

export default function Example() {
  return (
    <Divider variant="neutral" />
  )
}
```

## Ejemplos

### Neutral

Línea horizontal para fondos blancos o neutros. Color `#DEDEDE` light / `#333333` dark. Hairline = `0.5px`, thin = `1px`.

    <div style={{ width: '100%', height: '0.5px', backgroundColor: 'var(--preview-border)' }} />
    <div style={{ fontSize: '14px', fontFamily: 'var(--font-ubuntu-sans), sans-serif', color: 'inherit', marginTop: '16px' }}>Sección inferior</div>

```tsx title="divider-neutral.tsx"
export function DividerNeutral() {
  return <hr className="border-t border-[#DEDEDE] my-4" />;
}
```

```tsx title="divider-label.tsx"
export function DividerWithLabel({ label }: { label: string }) {
  return (
    <div className="flex items-center">
      <div className="flex-1 h-px bg-[#DEDEDE]" />
      <span className="px-3 text-xs text-[#999999]">{label}</span>
      <div className="flex-1 h-px bg-[#DEDEDE]" />
    </div>
  );
}
```

## Guías de Uso

<DoDont type="do">
  Usar el tono warm en fondos #FAF8F4, neutral en fondos blancos.
</DoDont>

<DoDont type="dont">
  No abusar de dividers, usar spacing suficiente antes de recurrir a líneas.
</DoDont>
