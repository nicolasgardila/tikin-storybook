```tsx title="pagination-default.tsx"
export function Pagination({ currentPage = 1, totalPages = 5, onPageChange }) {
  return (
    <div className="flex items-center gap-1 font-['Ubuntu_Sans'] text-sm">
      <button
        onClick={() => onPageChange(currentPage - 1)}
        disabled={currentPage === 1}
        className="w-9 h-9 flex items-center justify-center rounded-lg hover:bg-[#F6F6F6]"
      >
        <ChevronLeft size={16} />
      </button>
      {Array.from({ length: totalPages }, (_, i) => i + 1).map((page) => (
        <button
          key={page}
          onClick={() => onPageChange(page)}
          className={`w-9 h-9 flex items-center justify-center rounded-lg font-medium ${
            page === currentPage
              ? 'bg-[#0B0B0B] text-white'
              : 'text-[#0B0B0B] hover:bg-[#F6F6F6]'
          }`}
        >
          {page}
        </button>
      ))}
      <button
        onClick={() => onPageChange(currentPage + 1)}
        disabled={currentPage === totalPages}
        className="w-9 h-9 flex items-center justify-center rounded-lg hover:bg-[#F6F6F6]"
      >
        <ChevronRight size={16} />
      </button>
    </div>
  );
}
```

```tsx title="pagination-ellipsis.tsx"
export function PaginationWithEllipsis({ currentPage = 9, totalPages = 10, onPageChange }) {
  // Genera rango con ellipsis: [1, 2, '...', 8, 9, 10]
  const range = buildPageRange(currentPage, totalPages, { siblingCount: 1 });

  return (
    <div className="flex items-center gap-1 font-['Ubuntu_Sans'] text-sm">
      <button onClick={() => onPageChange(currentPage - 1)} className="w-9 h-9 flex items-center justify-center rounded-lg hover:bg-[#F6F6F6]">
        <ChevronLeft size={16} />
      </button>
      {range.map((item, i) =>
        item === '...' ? (
          <span key={i} className="w-9 h-9 flex items-center justify-center text-[#999999]">...</span>
        ) : (
          <button key={i} onClick={() => onPageChange(item)} className={`w-9 h-9 flex items-center justify-center rounded-lg font-medium ${item === currentPage ? 'bg-[#0B0B0B] text-white' : 'text-[#0B0B0B] hover:bg-[#F6F6F6]'}`}>
            {item}
          </button>
        )
      )}
      <button onClick={() => onPageChange(currentPage + 1)} className="w-9 h-9 flex items-center justify-center rounded-lg hover:bg-[#F6F6F6]">
        <ChevronRight size={16} />
      </button>
    </div>
  );
}
```

```tsx title="pagination-simple.tsx"
export function SimplePagination({ currentPage = 3, totalPages = 10, onPageChange }) {
  return (
    <div className="flex items-center gap-2 font-['Ubuntu_Sans'] text-sm">
      <button
        onClick={() => onPageChange(currentPage - 1)}
        disabled={currentPage === 1}
        className="w-9 h-9 flex items-center justify-center rounded-lg border border-[#DEDEDE] hover:bg-[#F6F6F6]"
      >
        <ChevronLeft size={16} />
      </button>
      <span className="font-medium min-w-[80px] text-center">
        Pagina {currentPage} de {totalPages}
      </span>
      <button
        onClick={() => onPageChange(currentPage + 1)}
        disabled={currentPage === totalPages}
        className="w-9 h-9 flex items-center justify-center rounded-lg border border-[#DEDEDE] hover:bg-[#F6F6F6]"
      >
        <ChevronRight size={16} />
      </button>
    </div>
  );
}
```

### Interacciones

El componente incluye feedback tactil al presionar: escala sutil (`scale 0.97`) con reduccion de opacidad, y un release con transicion spring natural. Este efecto se desactiva automaticamente en estados `disabled` y `loading`.
