## Instalacion

```bash title="Terminal"
npm install @dstikin/ds
```

```tsx
import { Avatar } from '@dstikin/ds'
```

## Uso

```tsx title="avatar.tsx"
import { Avatar } from '@dstikin/ds'

export default function Example() {
  return (
    <Avatar initials="NG" size="40" showStatus status="online" />
  )
}
```

```tsx title="avatar-initials.tsx"
export function AvatarInitials({ initials, size = 40 }: { initials: string; size?: number }) {
  const fontSizes: Record<number, number> = { 24: 9, 32: 11, 40: 14, 48: 16, 64: 22 };
  return (
    <div
      style={{
        width: size, height: size, fontSize: fontSizes[size],
        borderRadius: '50%', backgroundColor: '#E8E4DD', border: '0.5px solid #DEDEDE',
        display: 'flex', alignItems: 'center', justifyContent: 'center',
        fontFamily: 'var(--font-ubuntu-sans), sans-serif', fontWeight: 600,
        color: 'var(--preview-brand, #0B0B0B)', textTransform: 'uppercase', lineHeight: 1,
      }}
    >
      {initials}
    </div>
  );
}
```

```tsx title="avatar-status.tsx"
export function AvatarWithStatus({ initials, status }: { initials: string; status: 'online' | 'offline' | 'busy' | 'away' }) {
  const statusColors: Record<string, string> = { online: '#14AE5C', offline: '#B3B3B3', busy: '#EC221F', away: '#E8B931' };
  return (
    <div style={{ position: 'relative', display: 'inline-flex' }}>
      <div style={{ width: 48, height: 48, borderRadius: '50%', backgroundColor: '#E8E4DD', border: '0.5px solid #DEDEDE', display: 'flex', alignItems: 'center', justifyContent: 'center', fontFamily: 'var(--font-ubuntu-sans), sans-serif', fontWeight: 600, fontSize: 16, color: 'var(--preview-brand, #0B0B0B)', textTransform: 'uppercase', overflow: 'hidden' }}>
        {initials}
      </div>
      <div style={{ position: 'absolute', bottom: 1, right: 1, width: 12, height: 12, borderRadius: '50%', backgroundColor: statusColors[status], border: '2px solid #FAFAFA' }} />
    </div>
  );
}
```

## Guías de Uso

<DoDont type="do">
  Siempre proporcionar initials como fallback cuando la imagen puede no cargar.
</DoDont>

<DoDont type="dont">
  No usar avatares menores a 24px, la legibilidad se pierde.
</DoDont>
