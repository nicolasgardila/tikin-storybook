```tsx title="badge-numeric.tsx"
export function BadgeNumeric({ count, max = 99 }: { count: number; max?: number }) {
  const display = count > max ? `${max}+` : String(count);
  return (
    <span style={{ minWidth: 16, height: 16, borderRadius: 8, backgroundColor: 'var(--preview-brand)', color: 'var(--preview-brand-on)', fontSize: 10, fontFamily: 'var(--font-ubuntu-sans-mono), monospace', fontWeight: 500, display: 'inline-flex', alignItems: 'center', justifyContent: 'center', padding: '0 4px', lineHeight: 1 }}>
      {display}
    </span>
  );
}
```

## Guias de Uso

<DoDont type="do">
  Usa Badge para conteos de notificaciones, mensajes no leidos, o indicadores de atencion sobre iconos.
</DoDont>

<DoDont type="dont">
  No uses Badge para etiquetas de estado estaticas. Usa Status Chips en su lugar.
</DoDont>

<DoDont type="do">
  Usa el variant "dot" cuando solo necesitas indicar la existencia de novedades sin un conteo especifico.
</DoDont>

<DoDont type="dont">
  No muestres badges con conteo 0 a menos que sea necesario por contexto. Usa showZero solo si el cero tiene significado.
</DoDont>
