export function Table() {
  return (
    <div className="w-full rounded-xl overflow-hidden border border-[#DEDEDE]">
      <table className="w-full border-collapse font-['Ubuntu_Sans'] text-sm">
        <thead>
          <tr className="bg-[#F6F6F6]">

            <th className="text-left p-[12px_16px] font-semibold border-b border-[#DEDEDE]">Correo</th>
            <th className="text-right p-[12px_16px] font-semibold border-b border-[#DEDEDE]">Monto</th>
          </tr>
        </thead>
        <tbody>
          {data.map((row, i) => (
            <tr key={i} className={i % 2 === 0 ? 'bg-white' : 'bg-[#F6F6F6]'}>
              <td className="p-[12px_16px] border-b border-[#DEDEDE]">{row.nombre}</td>
              <td className="p-[12px_16px] border-b border-[#DEDEDE]">{row.correo}</td>
              <td className="p-[12px_16px] border-b border-[#DEDEDE] text-right">{row.monto}</td>

      </table>
    </div>
  );
}

## Guias de Uso

<DoDont type="do">
  Usa alineacion izquierda para texto y derecha para numeros. Esto facilita la lectura y comparacion de datos.
</DoDont>

<DoDont type="dont">
  No uses tablas para datos con 1-2 columnas simples, considera usar una lista o cards en su lugar.
</DoDont>
