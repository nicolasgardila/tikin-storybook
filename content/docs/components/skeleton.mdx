## Instalacion

```bash title="Terminal"
npm install @dstikin/ds
```

```tsx
import { Skeleton } from '@dstikin/ds'
```

## Uso

```tsx title="skeleton.tsx"
import { Skeleton } from '@dstikin/ds'

export default function Example() {
  return (
    <Skeleton variant="text" width="200px" height="16px" />
  )
}
```

## Ejemplos

### Líneas de Texto

Rectángulos con animación de pulso que simulan líneas de texto en carga.

<ComponentPreview>
  <div style={{ display: 'flex', flexDirection: 'column', gap: '8px', width: '100%' }}>
    <div style={{ width: '100%', height: '14px', backgroundColor: 'var(--preview-muted-bg)', borderRadius: '4px', animation: 'ds-pulse 1.8s ease-in-out infinite' }} />
    <div style={{ width: '80%', height: '14px', backgroundColor: 'var(--preview-muted-bg)', borderRadius: '4px', animation: 'ds-pulse 1.8s ease-in-out infinite', animationDelay: '0.15s' }} />
    <div style={{ width: '60%', height: '14px', backgroundColor: 'var(--preview-muted-bg)', borderRadius: '4px', animation: 'ds-pulse 1.8s ease-in-out infinite', animationDelay: '0.3s' }} />
  </div>
</ComponentPreview>

```tsx title="skeleton-text.tsx"
export function SkeletonText({ lines = 3 }: { lines?: number }) {
  const widths = ['100%', '80%', '60%'];
  return (
    <div className="flex flex-col gap-2">
      {Array.from({ length: lines }).map((_, i) => (
        <div
          key={i}
          className="animate-pulse"
          style={{ width: widths[i % widths.length], height: 14, backgroundColor: '#EEEBE6', borderRadius: 4 }}
        />
      ))}
    </div>
  );
}
```

```tsx title="skeleton-circle.tsx"
export function SkeletonCircle({ size = 40 }: { size?: number }) {
  return (
    <div
      className="rounded-full animate-pulse"
      style={{ width: size, height: size, backgroundColor: '#EEEBE6' }}
    />
  );
}
```

## Guías de Uso

<DoDont type="do">
  Usar skeletons que reflejen la estructura del contenido real que reemplazarán.
</DoDont>

<DoDont type="dont">
  No usar skeletons para contenido que carga instantáneamente (menos de 200ms).
</DoDont>
