```tsx title="snackbar-action.tsx"
export function SnackbarWithAction({ message, action, onAction }: SnackbarActionProps) {
  return (
    <div className="fixed bottom-6 left-1/2 -translate-x-1/2 z-[60] rounded-2xl bg-[#1E1E1E] p-1 shadow-lg min-w-[280px] max-w-[420px]">

        <button onClick={onAction} className="font-mono text-[13px] font-medium uppercase text-[#FDDC72] whitespace-nowrap">
          {action}
        </button>
      </div>
    </div>
  );
}
```

## Guias de Uso

<DoDont type="do">
  Usar para confirmaciones de acciones con opcion de deshacer, como eliminar o archivar elementos. Mantener mensajes breves y claros.
</DoDont>

<DoDont type="dont">
  No usar para mensajes criticos que requieren atencion obligatoria. Usar Alert o Dialog para esos casos.
</DoDont>
