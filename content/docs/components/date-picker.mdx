export function Calendar({ selected, today, onSelect }: CalendarProps) {
  return (
    <div className="bg-white border border-[#DEDEDE] rounded-xl p-4 w-[300px] shadow-md">
      <div className="flex justify-between items-center mb-4">
        <ChevronLeft className="w-5 h-5 cursor-pointer" />
        <span className="text-base font-semibold">Marzo 2026</span>
        <ChevronRight className="w-5 h-5 cursor-pointer" />
      </div>
      <div className="grid grid-cols-7 gap-0.5 text-center">
        {dayHeaders.map(d => (
          <div key={d} className="text-xs text-[#999999] font-medium py-1">{d}</div>
        ))}
        {days.map(day => (
          <div
            key={day}
            onClick={() => onSelect(day)}
            className={`w-9 h-9 flex items-center justify-center text-sm cursor-pointer rounded-full
              ${day === selected ? 'bg-[#0B0B0B] text-white' : ''}
              ${day === today ? 'border border-[#0B0B0B]' : ''}
              ${day !== selected ? 'hover:bg-[#F6F6F6]' : ''}
            `}
          >
            {day}
          </div>
        ))}
      </div>
    </div>
  );
}

```tsx title="date-picker-range.tsx"
export function DateRangePicker({ startDate, endDate, onSelect }: DateRangeProps) {
  return (
    <div className="bg-white border border-[#DEDEDE] rounded-xl p-4 w-[300px] shadow-md">
      {/* Same calendar grid with range highlighting */}
      {days.map(day => {
        const isEndpoint = day === startDate || day === endDate;
        const isInRange = day > startDate && day < endDate;
        return (
          <div className={`w-9 h-9 flex items-center justify-center text-sm
            ${isEndpoint ? 'bg-[#0B0B0B] text-white rounded-full' : ''}
            ${isInRange ? 'bg-[#F6F6F6]' : ''}
          `}>
            {day}
          </div>
        );
      })}
    </div>
  );
}
```

## Guias de Uso

<DoDont type="do">
  Mostrar el mes actual por defecto y resaltar la fecha de hoy para dar contexto temporal al usuario.
</DoDont>

<DoDont type="dont">
  No desactivar fechas sin indicar claramente por que no estan disponibles. Usar tooltips o texto explicativo.
</DoDont>
