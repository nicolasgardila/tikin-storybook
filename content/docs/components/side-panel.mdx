```tsx title="side-panel.tsx"
export function SidePanel({ open, onClose, title, desc, width = 464, position = 'right', children }) {
  if (!open) return null;

  return (
    <div className="fixed inset-0 z-50 bg-black/40" onClick={onClose}>
      {/* Outer shell — warm double-layer */}
      <div
        className={`absolute top-0 ${position === 'right' ? 'right-0' : 'left-0'} h-full rounded-[24px] border-[0.5px] border-[#E8E4DD] bg-[#F1EFEB] p-3`}
        style={{ width }}
        onClick={(e) => e.stopPropagation()}
      >
        {/* Close button 40x40 on outer */}
        <button onClick={onClose} className="absolute top-6 right-6 z-10 w-10 h-10 flex items-center justify-center">
          <XIcon size={16} />
        </button>
        {/* Inner panel */}
        <div className="h-full rounded-[20px] border-[0.5px] border-[#E8E4DD] bg-white overflow-hidden flex flex-col">
          <div className="px-6 pt-6">
            <h3 className="font-funnel-display text-xl font-normal pr-10">{title}</h3>
            {desc && <p className="text-sm text-black/70 mt-1">{desc}</p>}
            <div className="h-px bg-[#E8E4DD] mt-4 -mx-6 w-[calc(100%+48px)]" />
          </div>
          <div className="px-6 py-6 flex-1 overflow-auto">
            {children}
          </div>
        </div>
      </div>
    </div>
  );
}
```

## Guias de Uso

<DoDont type="do">
  Usa para contenido secundario que no requiere una pagina nueva (filtros, detalles, configuracion).
</DoDont>

<DoDont type="dont">
  No uses Side Panel para flujos principales o formularios largos que merecen su propia vista.
</DoDont>
