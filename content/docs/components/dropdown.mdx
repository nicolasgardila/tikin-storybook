```tsx title="dropdown-usage.tsx"
export function Dropdown({ trigger, items, onSelect, align = 'left' }) {
  const [open, setOpen] = useState(false);

  return (
    <div className="relative inline-block">
      <button onClick={() => setOpen(!open)}>
        {trigger}
      </button>

      {open && (
        <div className={cn(
          "absolute mt-1 min-w-[180px] rounded-xl border border-[#DEDEDE] bg-white py-1 shadow-md",
          align === 'right' ? 'right-0' : 'left-0'
        )}>
          {items.map((item) => (
            <button
              key={item.id}
              onClick={() => { onSelect(item.id); setOpen(false); }}
              className="w-full px-3 py-2 text-left text-sm hover:bg-[#F6F6F6]"
            >
              {item.label}
            </button>
          ))}
        </div>
      )}
    </div>
  );
}
```

## Guías de Uso

<DoDont type="do">
  Usa Dropdown para menús de acciones contextuales y listas de opciones cortas (menos de 7 items).
</DoDont>

<DoDont type="dont">
  No uses Dropdown para navegación principal. Usa sidebar o tabs en su lugar.
</DoDont>

<DoDont type="do">
  Agrupa items relacionados con divisores y resalta acciones destructivas en rojo.
</DoDont>

<DoDont type="dont">
  No anides dropdowns dentro de otros dropdowns. Simplifica la jerarquía de navegación.
</DoDont>
