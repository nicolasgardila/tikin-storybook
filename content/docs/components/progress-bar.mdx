```tsx title="progress-bar.tsx"
export function ProgressBar({ value }: { value: number }) {
  return (
    <div style={{ width: '100%', height: 6, backgroundColor: '#EEEBE6', borderRadius: 100, overflow: 'hidden' }}>
      <div
        style={{ height: '100%', backgroundColor: 'var(--preview-brand, #0B0B0B)', borderRadius: 100, transition: 'width 300ms ease', width: `${Math.min(100, Math.max(0, value))}%` }}
      />
    </div>
  );
}
```

```tsx title="step-indicator.tsx"
export function StepIndicator({ currentStep, totalSteps, labels }: {
  currentStep: number;
  totalSteps: number;
  labels?: string[];
}) {
  return (
    <div>
      <div className="flex items-center justify-center">
        {Array.from({ length: totalSteps }).map((_, i) => (
          <Fragment key={i}>
            {i > 0 && (
              <div className={`flex-1 max-w-[80px] h-0.5 ${i < currentStep ? 'bg-[#0B0B0B]' : 'bg-[#DEDEDE]'}`} />
            )}
            <div
              className={`w-3 h-3 rounded-full ${
                i < currentStep - 1
                  ? 'bg-[#0B0B0B]'
                  : i === currentStep - 1
                  ? 'bg-[#0B0B0B] outline outline-2 outline-offset-2 outline-[#0B0B0B]'
                  : 'bg-[#F6F6F6] border-[1.5px] border-[#DEDEDE]'
              }`}
            />
          </Fragment>
        ))}
      </div>
      {labels && (
        <div className="flex justify-between mt-2">
          {labels.map((label, i) => (
            <span
              key={i}
              className={`text-[10px] text-center flex-1 ${
                i <= currentStep - 1 ? 'text-[#0B0B0B]' : 'text-[#999999]'
              } ${i === currentStep - 1 ? 'font-semibold' : ''}`}
            >
              {label}
            </span>
          ))}
        </div>
      )}
    </div>
  );
}
```

## Guías de Uso

<DoDont type="do">
  Usar progress bar para operaciones con progreso medible, step indicator para flujos con pasos definidos.
</DoDont>

<DoDont type="dont">
  No usar progress bar para flujos de duración indeterminada (usar Skeleton o spinner).
</DoDont>
