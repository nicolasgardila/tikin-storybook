```tsx title="accordion-collapsed.tsx"
export function AccordionItem({ title, content, isOpen, onToggle }) {
  return (
    <div className="border-b border-[#DEDEDE]">
      <button
        className="flex w-full items-center justify-between p-4"
        onClick={onToggle}
      >
        <span className="text-base font-medium text-[#0B0B0B]">{title}</span>
        <ChevronIcon
          className={`h-4 w-4 text-[#999999] transition-transform duration-200 ${
            isOpen ? 'rotate-180' : ''
          }`}
        />
      </button>
      {isOpen && (
        <div className="px-4 pb-4 text-sm text-[#666666]">{content}</div>
      )}
    </div>
  );
}
```

```tsx title="accordion-group.tsx"
export function AccordionGroup() {
  const [openIndex, setOpenIndex] = useState(0);

  const items = [
    {
      title: "Como activo mi tarjeta?",
      content: "Abre la app, ve a la seccion Tarjetas y selecciona Activar tarjeta.",
    },
    {
      title: "Como veo mis movimientos?",
      content: "En la pantalla principal, desliza hacia abajo para ver tus movimientos recientes.",
    },
    {
      title: "Como contacto soporte?",
      content: "Ve a Configuracion > Ayuda > Contactar soporte.",
    },
  ];

  return (
    <div>
      {items.map((item, i) => (
        <AccordionItem
          key={i}
          title={item.title}
          content={item.content}
          isOpen={openIndex === i}
          onToggle={() => setOpenIndex(openIndex === i ? -1 : i)}
        />
      ))}
    </div>
  );
}
```

### Interacciones

El componente incluye feedback tactil al presionar: escala sutil (`scale 0.97`) con reduccion de opacidad, y un release con transicion spring natural. Este efecto se desactiva automaticamente en estados `disabled` y `loading`.

## Guias de Uso

<DoDont type="do">
  Usa para FAQs, configuraciones agrupadas, o contenido largo que se puede segmentar.
</DoDont>

<DoDont type="dont">
  No ocultes informacion critica dentro de un accordion que el usuario debe ver siempre.
</DoDont>
