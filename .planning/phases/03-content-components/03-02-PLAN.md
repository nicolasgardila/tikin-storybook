    1. Create `components/docs/do-dont.tsx`:

       ```tsx
       import type { ReactNode } from 'react';

       interface DoDontProps {
         type: 'do' | 'dont';
         children: ReactNode;
       }

       export function DoDont({ type, children }: DoDontProps) {
         const isDo = type === 'do';

         return (
           <div
             className={`my-4 rounded-lg border-l-4 p-4 ${
               isDo
                 ? 'border-green-500 bg-green-50 dark:bg-green-950/20'
                 : 'border-red-500 bg-red-50 dark:bg-red-950/20'
             }`}
           >
             <div className="mb-2 flex items-center gap-2">
               {isDo ? (
                 <svg className="h-5 w-5 text-green-600 dark:text-green-400" viewBox="0 0 20 20" fill="currentColor">
                   <path fillRule="evenodd" d="M16.707 5.293a1 1 0 010 1.414l-8 8a1 1 0 01-1.414 0l-4-4a1 1 0 011.414-1.414L8 12.586l7.293-7.293a1 1 0 011.414 0z" clipRule="evenodd" />
                 </svg>
               ) : (
                 <svg className="h-5 w-5 text-red-600 dark:text-red-400" viewBox="0 0 20 20" fill="currentColor">
                   <path fillRule="evenodd" d="M4.293 4.293a1 1 0 011.414 0L10 8.586l4.293-4.293a1 1 0 111.414 1.414L11.414 10l4.293 4.293a1 1 0 01-1.414 1.414L10 11.414l-4.293 4.293a1 1 0 01-1.414-1.414L8.586 10 4.293 5.707a1 1 0 010-1.414z" clipRule="evenodd" />
                 </svg>
               )}
               <span className={`text-sm font-semibold ${
                 isDo ? 'text-green-700 dark:text-green-300' : 'text-red-700 dark:text-red-300'
               }`}>
                 {isDo ? 'Do' : "Don't"}
               </span>
             </div>
             <div className="text-sm text-[var(--docs-fg)]">
               {children}
             </div>
           </div>
         );
       }
       ```

    2. Usage in MDX:
       ```mdx
       <DoDont type="do">
         Use semantic color tokens (`--ds-warm-bg`) instead of hardcoded hex values.
       </DoDont>

       <DoDont type="dont">
         Don't use hardcoded colors like `#F1EFEB` directly in component styles.
       </DoDont>
       ```

    The component uses:
    - Green left border + green bg for "Do" cards
    - Red left border + red bg for "Don't" cards
    - SVG checkmark/X icons (no external dependency)
    - Dark mode support with Tailwind dark: variants
    - `--docs-fg` for body text (theme-aware)

       interface StatusBadgeProps {
         status: Status;
       }

       const statusConfig: Record<Status, { label: string; className: string }> = {
         alpha: {
           label: 'Alpha',
           className: 'bg-amber-100 text-amber-800 dark:bg-amber-900/30 dark:text-amber-300',
         },
         beta: {
           label: 'Beta',
           className: 'bg-blue-100 text-blue-800 dark:bg-blue-900/30 dark:text-blue-300',
         },
         stable: {
           label: 'Stable',
           className: 'bg-green-100 text-green-800 dark:bg-green-900/30 dark:text-green-300',
         },
       };

       export function StatusBadge({ status }: StatusBadgeProps) {
         const config = statusConfig[status];

         return (
           <span
             className={`inline-flex items-center rounded-full px-2.5 py-0.5 text-xs font-medium ${config.className}`}
           >
             {config.label}
           </span>
         );
       }
