    1. Update `mdx-components.tsx` to use fumadocs-ui CodeBlock and Pre components:

       ```tsx
       import defaultComponents from 'fumadocs-ui/mdx';
       import type { MDXComponents } from 'mdx/types';
       import { CodeBlock, Pre } from 'fumadocs-ui/components/codeblock';

       export function getMDXComponents(components?: MDXComponents): MDXComponents {
         return {
           ...defaultComponents,
           pre: ({ ref: _ref, ...props }) => (
             <CodeBlock {...props}>
               <Pre>{props.children}</Pre>
             </CodeBlock>
           ),
           ...components,
         };
       }
       ```

    2. This replaces the default `<pre>` element with fumadocs-ui's CodeBlock which provides:
       - Copy-to-clipboard button
       - Title/filename display (via `title="filename.tsx"` in markdown fences)
       - Icon support
       - Shiki syntax highlighting (already configured by fumadocs-mdx)

    IMPORTANT: The `ref: _ref` destructuring is necessary because HTML `ref` attribute conflicts with React `forwardRef`. This is documented in fumadocs-ui.
  </action>

  <files>components/docs/props-table.tsx, mdx-components.tsx</files>
  <action>
    1. Create `components/docs/props-table.tsx`:

       ```tsx
       interface Prop {
         name: string;
         type: string;
         default?: string;
         description: string;
         required?: boolean;
       }

       interface PropsTableProps {
         props: Prop[];
       }

       export function PropsTable({ props }: PropsTableProps) {
         return (
           <div className="my-6 overflow-x-auto">
             <table className="w-full text-sm">
               <thead>
                 <tr className="border-b border-[var(--docs-border)]">
                   <th className="py-2 pr-4 text-left font-semibold">Prop</th>
                   <th className="py-2 pr-4 text-left font-semibold">Type</th>
                   <th className="py-2 pr-4 text-left font-semibold">Default</th>
                   <th className="py-2 text-left font-semibold">Description</th>
                 </tr>
               </thead>
               <tbody>
                 {props.map((prop) => (
                   <tr key={prop.name} className="border-b border-[var(--docs-border)]/50">
                     <td className="py-2 pr-4 font-mono text-[var(--ds-brand)]">
                       {prop.name}
                       {prop.required && <span className="ml-1 text-red-500">*</span>}
                     </td>
                     <td className="py-2 pr-4 font-mono text-[var(--docs-muted)]">{prop.type}</td>
                     <td className="py-2 pr-4 font-mono text-[var(--docs-muted)]">
                       {prop.default ?? '—'}
                     </td>
                     <td className="py-2 text-[var(--docs-fg)]">{prop.description}</td>
                   </tr>
                 ))}
               </tbody>
             </table>
           </div>
         );
       }
       ```

    2. Register PropsTable in `mdx-components.tsx`:

       ```tsx

       PropsTable,
       ```

    3. The component:

       - Uses `--docs-*` CSS variables for theming (works in dark/light mode)
       - Uses `--ds-brand` for prop names (matches DS branding)
       - Shows required indicator (*) for required props
       - Monospace font for prop names and types
       - Responsive with horizontal scroll on narrow screens

    IMPORTANT: Use CSS variables from our token system, not hardcoded colors. This ensures dark mode compatibility.
  </action>
