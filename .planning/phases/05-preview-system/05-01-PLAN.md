    1. Create `components/docs/component-preview.tsx` as a client component:

       ```tsx
       'use client';

       import { useState, type ReactNode } from 'react';

       interface ComponentPreviewProps {
         children: ReactNode;
       }

       export function ComponentPreview({ children }: ComponentPreviewProps) {
         const [viewport, setViewport] = useState<'desktop' | 'mobile'>('desktop');
         const [theme, setTheme] = useState<'light' | 'dark'>('light');

         return (
           <div className="my-6 overflow-hidden rounded-lg border border-[var(--docs-border)]">
             {/* Toolbar */}
             <div className="flex items-center gap-1 border-b border-[var(--docs-border)] bg-[var(--docs-bg)] px-3 py-2">
               {/* Viewport toggles */}
               <div className="flex items-center gap-1 rounded-md bg-[var(--docs-border)]/30 p-0.5">
                 <button
                   onClick={() => setViewport('desktop')}
                   className={`rounded px-2.5 py-1 text-xs font-medium transition-colors ${
                     viewport === 'desktop'
                       ? 'bg-[var(--docs-bg)] text-[var(--docs-fg)] shadow-sm'
                       : 'text-[var(--docs-muted)] hover:text-[var(--docs-fg)]'
                   }`}
                 >
                   {/* Desktop icon - monitor SVG */}
                   <svg className="inline-block h-3.5 w-3.5 mr-1" viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2">
                     <rect x="2" y="3" width="20" height="14" rx="2" />
                     <line x1="8" y1="21" x2="16" y2="21" />
                     <line x1="12" y1="17" x2="12" y2="21" />
                   </svg>
                   Desktop
                 </button>
                 <button
                   onClick={() => setViewport('mobile')}
                   className={`rounded px-2.5 py-1 text-xs font-medium transition-colors ${
                     viewport === 'mobile'
                       ? 'bg-[var(--docs-bg)] text-[var(--docs-fg)] shadow-sm'
                       : 'text-[var(--docs-muted)] hover:text-[var(--docs-fg)]'
                   }`}
                 >
                   {/* Mobile icon - smartphone SVG */}
                   <svg className="inline-block h-3.5 w-3.5 mr-1" viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2">
                     <rect x="5" y="2" width="14" height="20" rx="2" />
                     <line x1="12" y1="18" x2="12" y2="18" />
                   </svg>
                   Mobile
                 </button>
               </div>

               {/* Spacer */}
               <div className="flex-1" />

               {/* Theme toggle */}
               <button
                 onClick={() => setTheme(t => t === 'light' ? 'dark' : 'light')}
                 className="rounded p-1.5 text-[var(--docs-muted)] hover:text-[var(--docs-fg)] transition-colors"
                 title={`Switch to ${theme === 'light' ? 'dark' : 'light'} preview`}
               >
                 {theme === 'light' ? (
                   /* Sun icon */
                   <svg className="h-4 w-4" viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2">
                     <circle cx="12" cy="12" r="5" />
                     <line x1="12" y1="1" x2="12" y2="3" />
                     <line x1="12" y1="21" x2="12" y2="23" />
                     <line x1="4.22" y1="4.22" x2="5.64" y2="5.64" />
                     <line x1="18.36" y1="18.36" x2="19.78" y2="19.78" />
                     <line x1="1" y1="12" x2="3" y2="12" />
                     <line x1="21" y1="12" x2="23" y2="12" />
                     <line x1="4.22" y1="19.78" x2="5.64" y2="18.36" />
                     <line x1="18.36" y1="5.64" x2="19.78" y2="4.22" />
                   </svg>
                 ) : (
                   /* Moon icon */
                   <svg className="h-4 w-4" viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2">
                     <path d="M21 12.79A9 9 0 1 1 11.21 3 7 7 0 0 0 21 12.79z" />
                   </svg>
                 )}
               </button>
             </div>

             {/* Preview area — isolated from docs site theme */}
             <div
               className="flex items-center justify-center transition-colors duration-200"
               style={{
                 backgroundColor: theme === 'dark' ? '#0B0B0B' : '#FAF8F4',
                 color: theme === 'dark' ? '#FFFFFF' : '#0B0B0B',
                 padding: '2rem',
                 minHeight: '120px',
               }}
             >
               <div
                 className="w-full transition-all duration-200"
                 style={{
                   maxWidth: viewport === 'mobile' ? '375px' : '100%',
                 }}
               >
                 {children}
               </div>
             </div>
           </div>
         );
       }
       ```

       Key design decisions:
       - **`'use client'`** — needed for useState (viewport/theme toggles)
       - **Inline styles for preview background** — uses raw DS hex values (#FAF8F4 warm bg for light, #0B0B0B brand for dark) instead of CSS variables. This ensures preview theme is INDEPENDENT of docs site theme (which controls --docs-* variables).
       - **Desktop = full width, Mobile = 375px** centered with smooth transition
       - **Toolbar** uses --docs-* variables (inherits docs site theme) while preview area uses hardcoded DS values
       - **SVG icons** inline (no external dependency, consistent with other doc components)
       - **Active state** on toolbar buttons with background change and shadow
